142static inline int calculate_accurate_MBR(uint8_t sensor_num, int val)
143{ // for better accuracy, enlarge SDR to two byte scale
144 if (SDR_M(sensor_num) == 0) {
145 return ((val << 8) * SDR_Rexp(sensor_num));
146 }
147 return ((val << 8) / SDR_M(sensor_num) * SDR_Rexp(sensor_num));
148}
149
150static inline int calculate_MBR(uint8_t sensor_num, int val)
151{
152 if (SDR_M(sensor_num) == 0) {
153 return (val * SDR_Rexp(sensor_num) + round_add(sensor_num, val));
154 }
155 return (val * SDR_Rexp(sensor_num) / SDR_M(sensor_num) + round_add(sensor_num, val));
156}
157
158static inline float convert_MBR_to_reading(uint8_t sensor_num, uint8_t val)
159{
160 if (SDR_M(sensor_num) == 0) {
161 return (val - round_add(sensor_num, val)) / SDR_Rexp(sensor_num);
162 }
163
164 return (val - round_add(sensor_num, val)) * SDR_M(sensor_num) / SDR_Rexp(sensor_num);
165}
